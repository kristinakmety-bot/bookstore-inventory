    function updateRow(el, book) {

      el.querySelector('.book-name').textContent = book.name || 'Untitled';

      const metaBits = [

        book.author,

        book.condition ? `Condition: ${book.condition}` : '',

        book.language, book.genre, book.publisher,

        book.publishing_year ? `${book.publishing_year}` : '',

        book.isbn ? `ISBN: ${book.isbn}` : ''

      ].filter(Boolean);

      el.querySelector('.book-meta').textContent = metaBits.length ? metaBits.join(' · ') : '—';

      el.querySelector('.book-qty').textContent = String(book.quantity ?? 0);

      el.querySelector('.book-sold').textContent = String(book.sold_count ?? 0);

      el.querySelector('.book-purchase').textContent = fmt(book.purchase_price);

      const price = el.querySelector('.book-price');

      const tag = el.querySelector('.book-sale-tag');

      if (book.on_sale && Number(book.sale_price) > 0) {

        price.innerHTML = '<span style="text-decoration:line-through;color:#a99e88;font-weight:400;margin-right:6px;">' + fmt(book.sell_price) + '</span>' + fmt(book.sale_price);

        tag.classList.remove('hidden');

      } else {

        price.textContent = fmt(book.sell_price);

        tag.classList.add('hidden');

      }

    }

    function wireRow(el, id) {

      const defaultBox = el.querySelector('.actions-default');

      const sellBox = el.querySelector('.sell-controls');

      const delBox = el.querySelector('.delete-controls');


      el.querySelector('.book-edit').addEventListener('click', () => {

        const rec = books.find(b => b.__backendId === id);

        if (rec) { switchView('inventory'); openForm(rec); }

      });


      const discSel = sellBox.querySelector('.sell-discount');

      const customInput = sellBox.querySelector('.sell-custom');

      discSel.addEventListener('change', () => customInput.classList.toggle('hidden', discSel.value !== 'custom'));


      el.querySelector('.book-sell').addEventListener('click', () => {

        defaultBox.classList.add('hidden'); sellBox.classList.remove('hidden');

        const qi = sellBox.querySelector('.sell-qty'); qi.value = '1'; qi.focus();

        discSel.value = '0'; customInput.value = ''; customInput.classList.add('hidden');

      });

      el.querySelector('.book-cancel-sell').addEventListener('click', () => resetRowActions(el));

      el.querySelector('.book-confirm-sell').addEventListener('click', async () => {

        const rec = books.find(b => b.__backendId === id);

        if (!rec) return;

        const qi = sellBox.querySelector('.sell-qty');

        const amount = Math.max(1, parseInt(qi.value, 10) || 1);

        if ((Number(rec.quantity) || 0) < amount) { showToast('Not enough copies in stock.', true); return; }

        const pct = readDiscount(sellBox, '.sell-discount', '.sell-custom');

        const unit = effectivePrice(rec) * (1 - pct / 100);

        const total = unit * amount;

        const sp = sellBox.querySelector('.sell-spinner');

        sp.classList.remove('hidden');

        sellBox.querySelectorAll('button').forEach(b => b.disabled = true);

        rec.quantity = (Number(rec.quantity) || 0) - amount;

        rec.sold_count = (Number(rec.sold_count) || 0) + amount;

        const result = await window.dataSdk.update(rec);

        if (result.isOk) {

          const dtxt = pct > 0 ? ` at ${pct}% off (${fmt(total)})` : ` (${fmt(total)})`;

          showToast(`Sold ${amount} cop${amount === 1 ? 'y' : 'ies'}${dtxt}.`, false);

          // Log sales trend record

          await window.dataSdk.create({

            record_kind: 'sales_log',

            item_id: id,

            item_name: rec.name || 'Untitled',

            item_kind: kindOf(rec),

            sale_qty: amount,

            sold_at: new Date().toISOString()

          });

        } else {

          rec.quantity = (Number(rec.quantity) || 0) + amount;

          rec.sold_count = (Number(rec.sold_count) || 0) - amount;

          sp.classList.add('hidden');

          sellBox.querySelectorAll('button').forEach(b => b.disabled = false);

          showToast('Could not record the sale. Please try again.', true);

        }

      });


      el.querySelector('.book-delete').addEventListener('click', () => {

        defaultBox.classList.add('hidden'); delBox.classList.remove('hidden');

      });

      el.querySelector('.book-cancel').addEventListener('click', () => resetRowActions(el));

      el.querySelector('.book-confirm').addEventListener('click', async () => {

        const rec = books.find(b => b.__backendId === id);

        if (!rec) return;

        const cs = delBox.querySelector('.confirm-spinner');

        cs.classList.remove('hidden');

        delBox.querySelectorAll('button').forEach(b => b.disabled = true);

        const result = await window.dataSdk.delete(rec);

        if (result.isOk) { showToast('Book deleted.', false); }

        else {

          cs.classList.add('hidden');

          delBox.querySelectorAll('button').forEach(b => b.disabled = false);

          showToast('Could not delete the book. Please try again.', true);

        }

      });

    }


    function filteredBooks() {

      const q = (document.getElementById('f-search').value || '').trim().toLowerCase();

    function renderList() {

    }

    function wireBundleRow(el, id) {

      el.querySelector('.bundle-confirm-sell').addEventListener('click', async () => {

        const rec = books.find(b => b.__backendId === id);

        if (!rec) return;

        const qi = sellBox.querySelector('.bundle-sell-qty');

        const amount = Math.max(1, parseInt(qi.value, 10) || 1);


        let items = [];

        try { items = JSON.parse(rec.bundle_items || '[]'); } catch (e) { items = []; }

        // Verify each included item has enough stock

        const affected = [];

        for (const it of items) {

          const child = books.find(b => b.__backendId === it.id);

          if (!child) { showToast(`"${it.name || 'An item'}" is no longer in inventory.`, true); return; }

          if ((Number(child.quantity) || 0) < amount) { showToast(`Not enough stock of "${child.name}".`, true); return; }

          affected.push(child);

        }

        if ((Number(rec.quantity) || 0) < amount) { showToast('Not enough bundles available.', true); return; }


        const sp = sellBox.querySelector('.bundle-sell-spinner');

        sp.classList.remove('hidden');

        sellBox.querySelectorAll('button').forEach(b => b.disabled = true);


        // Decrement each child item

        const snapshot = affected.map(c => ({ ref: c, qty: c.quantity, sold: c.sold_count }));

        let failed = false;

        for (const child of affected) {

          child.quantity = (Number(child.quantity) || 0) - amount;

          child.sold_count = (Number(child.sold_count) || 0) + amount;

          const r = await window.dataSdk.update(child);

          if (!r.isOk) { failed = true; break; }

        }

        if (!failed) {

          rec.quantity = (Number(rec.quantity) || 0) - amount;

          rec.sold_count = (Number(rec.sold_count) || 0) + amount;

          const r = await window.dataSdk.update(rec);

          if (!r.isOk) failed = true;

        }


        if (!failed) {

          showToast(`Sold ${amount} bundle${amount === 1 ? '' : 's'} (${fmt((Number(rec.bundle_price) || 0) * amount)}).`, false);

          // Log sales trend record

          await window.dataSdk.create({

            record_kind: 'sales_log',

            item_id: id,

            item_name: rec.name || 'Untitled',

            item_kind: 'bundle',

            sale_qty: amount,

            sold_at: new Date().toISOString()

          });

        } else {

          // Roll back child changes

          snapshot.forEach(s => { s.ref.quantity = s.qty; s.ref.sold_count = s.sold; });

          sp.classList.add('hidden');

          sellBox.querySelectorAll('button').forEach(b => b.disabled = false);

          showToast('Could not complete the bundle sale. Please try again.', true);

        }

      });

        defaultBox.classList.add('hidden'); delBox.classList.remove('hidden');

      });

      el.querySelector('.bundle-cancel').addEventListener('click', () => resetBundleRowActions(el));

      el.querySelector('.bundle-confirm').addEventListener('click', async () => {

        const rec = books.find(b => b.__backendId === id);

        if (!rec) return;

        const cs = delBox.querySelector('.bundle-confirm-spinner');

        cs.classList.remove('hidden');

          showToast('Could not delete the bundle. Please try again.', true);

        }

    function renderTrendsGraph() {

      const now = new Date();

      const cutoffDate = new Date();

      if (selectedTrendsTimeframe === 'daily') {

        cutoffDate.setHours(0, 0, 0, 0); // Today only

      } else if (selectedTrendsTimeframe === 'weekly') {

        cutoffDate.setDate(now.getDate() - 7); // Last 7 days

      } else if (selectedTrendsTimeframe === 'monthly') {

        cutoffDate.setMonth(now.getMonth() - 1); // Last 30 days

      }


      // Aggregate quantities by product/item name

      const itemMap = new Map();

      salesLogs.forEach(log => {

        const logDate = new Date(log.sold_at || log.created_at);

        if (logDate >= cutoffDate) {

          const qty = Number(log.sale_qty) || 0;

          const name = log.item_name || 'Unknown Item';

          itemMap.set(name, (itemMap.get(name) || 0) + qty);

        emptyMsg.classList.remove('hidden');

        return;

      }

        .sort((a, b) => b[1] - a[1])

        .slice(0, 5);

      sortedItems.forEach(([name, qty]) => {

        col.innerHTML = `

          <div class="w-full bg-[#f4d68a] hover:bg-[#8a5a2b] rounded-t-md transition-all duration-300 relative" style="height: ${percent}%;">

            <div class="absolute -top-7 left-1/2 -translate-x-1/2 bg-[#3a2e1f] text-[#fff8ec] text-[9px] px-1.5 py-0.5 rounded shadow opacity-0 group-hover:opacity-100 transition-opacity whitespace-nowrap z-10">

              ${name}: ${qty} sold

            </div>

          <p class="text-[10px] text-[#6b5a41] truncate w-full text-center mt-2 font-medium" title="${name}">${name}</p>
